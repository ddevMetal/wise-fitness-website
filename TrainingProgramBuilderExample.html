Method 1: Direct HTML Include

Method 2: JavaScript Integration

// Load the builder dynamically function loadTrainingBuilder() { const script = document.createElement('script'); script.src = 'TrainingProgramBuilder.html'; document.head.appendChild(script); } // Open builder function openBuilder() { if (typeof openTrainingBuilder === 'function') { openTrainingBuilder(); } else { loadTrainingBuilder(); setTimeout(openTrainingBuilder, 500); } }

Method 3: Framework Integration (React/Vue/Angular)

// React Component Example import { useEffect } from 'react'; function TrainingProgramButton() { useEffect(() => { // Load the training builder script const script = document.createElement('script'); script.src = '/TrainingProgramBuilder.html'; document.head.appendChild(script); return () => { document.head.removeChild(script); }; }, []); const handleClick = () => { if (window.openTrainingBuilder) { window.openTrainingBuilder(); } }; return ( <button onClick={handleClick}> Create Training Program </button> ); }
